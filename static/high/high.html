#define YDWETimerSystemIncluded

library YDWETimerPeriodic initializer Init requires YDWEBase
    globals
        private integer array Structs   //中心计时器存储队列
        private constant timer TIMER22 = CreateTimer()    //创建时间滚动计时器
        private constant real  INTER = .01             //中心计时器检查精度
    endglobals

    //define定义伪函数方便使用
    #define Timer_GetTime()     Timer_CurrentTime
    #define Timer_GetIndex()    Timer_CurrentIndex
    #define Timer_GetRemaining(t)   (Timer_Data[t]-Timer_CurrentTime)  //剩余时间
    #define Timer_SetTimer(t,t2)   set Timer_Data[t]=Timer_CurrentTime+t2
    #define Timer_GetTimer(t)   Timer_Data[t]

    struct encap
        real a = 0.0
        private real b = 0.0
        public real c = 4.5
        method randomize takes nothing returns nothing
            // 结构自己的成员可以访问所有其他成员
            set this.a = GetRandomReal(0, 45.0)
            set this.b = GetRandomReal(0, 45.0)
            set this.c = GetRandomReal(0, 45.0)
        endmethod
    endstruct

    private struct TimerData
        trigger trg   = null    //执行动作
        real    dur   = 0.      //逝去时间
        real    inter = 0.      //到期时间
        boolean perio = false   //是否循环
        boolean pause = false   //是否执行
    endstruct

    private function Spin takes nothing returns nothing
        local TimerData data = 0
        local integer   inst = 0
        loop
            exitwhen inst == Total      //直到最后一个中心计时器
            set data = inst:Structs     //获取当前对象的索引
            set data.dur = data.dur + INTER     //当然时间增加
            if data.dur == data.inter or data.dur > data.inter then     //时间到期 计时器逝去时间大于等到期时间
                set Index = integer(data)       //设置全局索引保存当前所有
                if not data.pause then          //是否 不暂停
                    call TriggerExecute(data.trg)   //不暂停执行动作
                endif
                if data.perio then      //是否循环
                    set data.dur = 0.   //清空计时
                else
                    set data.trg = null     //42-43 销毁计时器
                    call data.destroy()     //销毁对象
                    
                    set Total = Total - 1   //总数减一
                    set inst:Structs = Total:Structs    //最后一个计时器移动到当前位置
                    set inst = inst - 1     //检查位置到上一个 使下一次检查当前位置保存的最后一个
                endif
                set Index = 0
            endif
            set inst = inst + 1         //检查下一个
        endloop
        if Total == 0 then      //队列为空
            call PauseTimer(TIMER)      //暂停周期滚动检查
        endif
    endfunction

    function YDWE_TP_NewTimerPeriodic takes real timeout, boolean periodic, trigger trig returns TimerData
        local TimerData data = TimerData.create()   //创建新对象
        set data.inter = timeout        //89-93 初始化数据
        set data.perio = periodic
        set data.trg   = trig
        set Total:Structs = data
        set Total = Total + 1 
        if Total - 1 == 0 then          //加入第一个则启动滚动处理
            call TimerStart(TIMER, INTER, true, function Spin)
        endif
        //call BJDebugMsg(I2S(Total))
        return data         //返回新建中心计时器索引
    endfunction

endlibrary
